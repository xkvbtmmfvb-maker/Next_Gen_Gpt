class BaseDefinition:
"""A generic base for both Blueprints and Manifestations."""
def __init__(self, name: str, description: str = ""):
self.id = str(uuid.uuid4())
self.name = name
self.description = description
self.properties = {}     # Key-value pairs describing attributes
self.relationships = {}  # Links to other Definitions (e.g., {"depends_on": [def_id, ...]})
self.history = []        # For memory - stores changes over time
self.current_state_snapshot = {} # To hold current state for history logging

def add_property(self, key: str, value):
self.properties[key] = value
self._update_current_state_snapshot()

def add_relationship(self, rel_type: str, target_id: str):
if rel_type not in self.relationships:
self.relationships[rel_type] = []
self.relationships[rel_type].append(target_id)
self._update_current_state_snapshot()

def _update_current_state_snapshot(self):
"""Internal helper to keep snapshot up-to-date. Subclasses will extend this."""
self.current_state_snapshot = self.properties.copy()

def update_history(self, change_description: str, current_state_snapshot: dict = None):
"""Records changes over time."""
self.history.append({
"timestamp": time.time(),
"tick": RealitySimulationEngine.current_tick, # Global tick from scheduler
"change": change_description,
"state": current_state_snapshot if current_state_snapshot is not None else self.current_state_snapshot.copy()
})

def get_semantic_content(self) -> str:
"""
Combines relevant text for embedding generation.
This method is crucial for the semantic programming layer.
"""
content = f"{self.name} {self.description}"
# Include properties in semantic content, convert values to string for simplicity
for k, v in self.properties.items():
content += f" {k}:{str(v)}"
return content

def __repr__(self):
return f"Def({self.name})"

class Scheduler:
"""

"""
def __init__(self, tick_interval_seconds: float = 0.1):

self.tick_interval = tick_interval_seconds
self._is_running = False
self.current_tick = 0
self.focused_sector: SimulationSector | None = None # NEW: for attention/focus

def start(self, engine_callback_func):

print(f"Scheduler started with initial tick interval: {self.tick_interval}s")

"""Dynamically adjusts the tick rate."""
if new_interval <= 0:
raise ValueError("Tick interval must be positive.")
if self.tick_interval != new_interval:
self.tick_interval = new_interval
print(f"  Scheduler: Tick interval adjusted to {self.tick_interval:.3f}s.")

"""Resets the tick interval to its initial value."""
if self.tick_interval != self._initial_tick_interval:
self.tick_interval = self._initial_tick_interval
print(f"  Scheduler: Tick interval reset to {self.tick_interval:.3f}s.")

"""Directs the scheduler's attention to a specific sector."""
if self.focused_sector != sector:
self.focused_sector = sector
if sector:
print(f"  Scheduler: Focusing on sector {sector.name}.")
else:

class EmbeddingModel:
"""
Conceptual embedding model. Replaces keyword matching with vector embeddings.
In a real system, this would be a large NLP model (e.g., Sentence Transformers, OpenAI embeddings).
For pseudocode, it's a simple hashing/keyword presence for demonstration purposes.
"""
def __init__(self):
# Placeholder for a loaded model or API client

def generate_embedding(self, text: str) -> list[float]:
"""
Generates a conceptual vector embedding for a given text.
In a real system, this would use a robust NLP model like:
from sentence_transformers import SentenceTransformer
model = SentenceTransformer('all-MiniLM-L6-v2')
return model.encode(text).tolist()

Here, a simple "pseudo-embedding" based on character counts, length,
and presence of common "conceptual" words to simulate distinct vectors.
"""
if not text: return [0.0] * 9 # Return a zero vector for empty text

# A very simplistic placeholder for vector generation
embedding_vec = [
sum(ord(char) for char in text) / 10000.0, # Magnitude based on chars
len(text) / 100.0,                         # Length
1.0 if "concept" in text.lower() else 0.0,
1.0 if "system" in text.lower() else 0.0,
1.0 if "rule" in text.lower() else 0.0,
1.0 if "design" in text.lower() else 0.0,
1.0 if "physical" in text.lower() else 0.0,
1.0 if "applied" in text.lower() else 0.0,
random.random() * 0.01 # Small random component for variation, to prevent perfect matches often
]
return embedding_vec

def cosine_similarity(self, vec1: list[float], vec2: list[float]) -> float:
"""
Calculates cosine similarity between two conceptual vectors.
(dot product / (norm1 * norm2)).
"""
if not vec1 or not vec2 or len(vec1) != len(vec2):
return 0.0

dot_product = sum(a * b for a, b in zip(vec1, vec2))
norm_vec1 = sum(a * a for a in vec1) ** 0.5
norm_vec2 = sum(b * b for b in vec2) ** 0.5

if norm_vec1 == 0 or norm_vec2 == 0:
return 0.0 # Handle zero vectors

return dot_product / (norm_vec1 * norm_vec2)


class SemanticMatcher:
"""
Uses vector embeddings for robust semantic similarity calculations.
Replaces keyword-based logic for a more nuanced "conceptual understanding."
"""
def __init__(self):
self.embedding_model = EmbeddingModel()
self._cache = {} # Cache embeddings to avoid re-computation for efficiency

def _get_embedding(self, definition: BaseDefinition) -> list[float]:
"""Get or generate embedding for a definition."""
if definition.id not in self._cache:
text_content = definition.get_semantic_content()
self._cache[definition.id] = self.embedding_model.generate_embedding(text_content)
return self._cache[definition.id]

def calculate_similarity(self, def1: BaseDefinition, def2: BaseDefinition) -> float:
"""
Calculates semantic similarity between two definitions using their embeddings.
"""
embedding1 = self._get_embedding(def1)
embedding2 = self._get_embedding(def2)
return self.embedding_model.cosine_similarity(embedding1, embedding2)

def find_best_match(self, source_def: BaseDefinition, target_definitions: list[BaseDefinition], min_similarity: float = 0.6) -> BaseDefinition | None:
"""Finds the best matching definition from a list based on semantic similarity."""
best_match = None
highest_similarity = min_similarity
for target_def in target_definitions:
similarity = self.calculate_similarity(source_def, target_def)
if similarity > highest_similarity:
highest_similarity = similarity
best_match = target_def
return best_match

def _evaluate_constraints(self, manifestations: list[Manifestation]):
"""Checks MP and HA constraints to maintain system integrity (the 'grammar')."""

for constraint in self.constraints.values():
if constraint.sector and constraint.sector != manifestation.sector:

if constraint.type == ConstraintType.DIRECTIVE_LOGIC:
if constraint.name == "Melting Point" and "temperature" in manifestation.physical_attributes:
current_temp = manifestation.physical_attributes["temperature"]
if current_temp >= constraint.value and manifestation.physical_attributes.get("material_state") != "liquid":
manifestation.update_physical_state("material_state", "liquid")
print(f"    MP '{constraint.name}' applied: '{manifestation.name}' melts at {current_temp}°F.")

# HA Constraint (Integrity/Condition) - Triggers anomalies/crashes
if constraint.type == ConstraintType.INTEGRITY_CONDITION:
if constraint.name == "Heart Attack Threshold" and "cardiac_stress" in manifestation.physical_attributes:
if manifestation.physical_attributes["cardiac_stress"] >= constraint.value:
error_msg = f"HA '{constraint.name}' violated! '{manifestation.name}' experiences critical biological crash!"
print(f"    ERROR: {error_msg}")
manifestation.update_physical_state("system_status", "CRITICAL_ERROR: BIOLOGICAL_CRASH")

self._trigger_anomaly_response(manifestation, constraint)

current_altitude = manifestation.physical_attributes["altitude"]
if current_altitude > constraint.value and manifestation.physical_attributes.get("alert_status") != "HIGH_ALTITUDE_CRISIS":
error_msg = f"HA '{constraint.name}' violated! '{manifestation.name}' exceeds critical altitude ({current_altitude}ft)!"
print(f"    WARNING: {error_msg}")
manifestation.update_physical_state("alert_status", "HIGH_ALTITUDE_CRISIS")
self._record_anomaly(manifestation, constraint, error_msg)
self._trigger_anomaly_response(manifestation, constraint)

def _evolve_existing_manifestations(self, manifestations: list[Manifestation]):
"""Simulates the dynamic changes in physical reality over time."""
print("  Evolving existing manifestations...")
for manifestation in manifestations:
# Example: A physical object experiencing physics (temperature)
if manifestation.sector in [SimulationSector.PHYSICS, SimulationSector.AVIATION]:
if "temperature" in manifestation.physical_attributes:
# Random fluctuation, or cooling if no heat source
if manifestation.physical_attributes.get("material_state") != "liquid":
manifestation.update_physical_state("temperature", manifestation.physical_attributes["temperature"] + random.uniform(-2, 5))
# Example: An aircraft flying
if manifestation.name.startswith("Airbus A380"):
current_altitude = manifestation.physical_attributes.get("altitude", 0)
speed = manifestation.physical_attributes.get("speed", 0)
if speed > 0:
manifestation.update_physical_state("altitude", current_altitude + speed / 100 * random.uniform(0.8, 1.2)) # Simulating climb/descend
if current_altitude > 35000: # Max cruising altitude usually
manifestation.update_physical_state("speed", max(0, speed + random.uniform(-10, 5))) # Speed might reduce at high alt
else:
manifestation.update_physical_state("speed", min(600, speed + random.uniform(-5, 10))) # Speed up during takeoff
else: # Grounded
manifestation.update_physical_state("altitude", 0)
manifestation.update_physical_state("speed", 0)

# Example: Patient's cardiac stress
if manifestation.name.startswith("Patient X"):
current_stress = manifestation.physical_attributes.get("cardiac_stress", 0)
manifestation.update_physical_state("cardiac_stress", max(0, min(150, current_stress + random.uniform(-10, 15))))
if manifestation.physical_attributes.get("system_status") == "CRITICAL_ERROR: BIOLOGICAL_CRASH":
manifestation.update_physical_state("health_status", "critical") # Update status if crashed
manifestation.update_physical_state("cardiac_stress", 200) # Keep stress high to reflect crisis

# Example: Economic manifestations
if manifestation.sector == SimulationSector.ECONOMICS:
current_growth = manifestation.physical_attributes.get("growth_rate", 0)
current_stability = manifestation.physical_attributes.get("market_stability", 0)
manifestation.update_physical_state("growth_rate", current_growth + random.uniform(-0.02, 0.03))
manifestation.update_physical_state("market_stability", max(0.1, min(1.0, current_stability + random.uniform(-0.05, 0.03))))

# Example: Ecological manifestations
if manifestation.sector == SimulationSector.ECOLOGY:
current_health = manifestation.physical_attributes.get("health_index", 1.0)
manifestation.update_physical_state("health_index", max(0.0, min(1.0, current_health + random.uniform(-0.05, 0.01))))

# --- NEW FEATURE: Intent Drift ---
def _apply_intent_drift(self):
"""
Blueprints slowly mutate based on outcomes (success/failure rates).
This is where the system begins to "learn" and adapt definitions based on "reality."
"""
print("  Applying intent drift to blueprints...")
# Use list(.) to iterate over a copy, as blueprint properties might change
for bp_id, blueprint in list(self.blueprints.items()):
# Ignore emergent patterns or blueprints specifically for recovery protocols
if blueprint.category == BlueprintCategory.EMERGENT_PATTERN or \
"RecoveryProtocolFor_" in blueprint.name:
continue

total_outcomes = blueprint.failure_count + blueprint.success_count
if total_outcomes > 5 and self.current_tick % 5 == 0: # Only start drifting after some significant interactions, and not every tick
failure_rate = blueprint.failure_count / total_outcomes
old_priority = blueprint.priority

# Example drift logic:
if failure_rate > 0.6: # If mostly failing
new_priority = max(0.1, round(blueprint.priority - 0.1, 2)) # Decrease priority, round to 2 decimals
if new_priority != old_priority:
blueprint.priority = new_priority
self._record_intent_drift(bp_id, "priority_decrease", old_priority, new_priority,
f"High failure rate ({failure_rate:.2f})")
print(f"    DRIFT: Blueprint '{blueprint.name}' priority decreased to {new_priority:.2f} due to high failure rate.")
# Add a property to indicate need for review/re-evaluation
if not blueprint.properties.get("needs_re_evaluation"):
blueprint.update_blueprint_property("needs_re_evaluation", True)
self._record_intent_drift(bp_id, "add_property", "None", "needs_re_evaluation:True",
f"High failure rate ({failure_rate:.2f})")
elif failure_rate < 0.2 and blueprint.success_count > 10: # If mostly succeeding
new_priority = min(1.0, round(blueprint.priority + 0.05, 2)) # Increase priority
if new_priority != old_priority:
blueprint.priority = new_priority
self._record_intent_drift(bp_id, "priority_increase", old_priority, new_priority,
f"High success rate ({(1-failure_rate):.2f})")
print(f"    DRIFT: Blueprint '{blueprint.name}' priority increased to {new_priority:.2f} due to high success rate.")
# Remove re-evaluation flag if it was there
if blueprint.properties.get("needs_re_evaluation"):
old_value = blueprint.properties["needs_re_evaluation"]
del blueprint.properties["needs_re_evaluation"]
blueprint._update_current_state_snapshot() # Ensure snapshot reflects removal
self._record_intent_drift(bp_id, "remove_property", f"needs_re_evaluation:{old_value}", "None",
f"Consistently successful ({(1-failure_rate):.2f})")


# --- NEW FEATURE: Feedback into the Scheduler (Time Awareness) ---
def _assess_system_load_and_focus(self):
"""
Adjusts scheduler parameters (tick rate, focus) based on system state.
This simulates the system's "attention" and "urgency" levels.
"""
print("  Assessing system load and focus...")
total_anomalies = len([a for a in self.historical_records["anomaly_logs"] if a["tick"] == self.current_tick])
critical_manifestations = [m for m in self.manifestations.values() if "CRITICAL_ERROR" in str(m.physical_attributes.get("system_status")) or "HIGH_ALTITUDE_CRISIS" == str(m.physical_attributes.get("alert_status"))]
high_priority_blueprints_pending = [b for b in self.blueprints.values() if b.priority > 0.8 and b.properties.get("manifested_status") != "complete"]
new_emergencies = [m for m in self.manifestations.values() if m.category == ManifestationCategory.OBSERVED_EMERGENCE and m.physical_attributes.get("status") == "critical" and m.history and m.history[-1]['tick'] == self.current_tick]

# 1. Adjust Tick Rate (Urgency)
# Speed up if there are critical issues or new emergencies
if total_anomalies > 0 or len(critical_manifestations) > 0 or len(new_emergencies) > 0:
if self.scheduler.tick_interval > 0.01: # Don't go below 0.01s
self.scheduler.adjust_tick_interval(max(0.01, self.scheduler.tick_interval * 0.8)) # Accelerate by 20%
# Slow down if the system is very large and calm, simulating "deep thought" or resource conservation
elif len(self.manifestations) > 50 and len(self.blueprints) > 50 and self.current_tick % 10 == 0: # Check every 10 ticks for large systems
if self.scheduler.tick_interval < 0.5: # Don't go above 0.5s
self.scheduler.adjust_tick_interval(min(0.5, self.scheduler.tick_interval * 1.2)) # Decelerate by 20%
else:
self.scheduler.reset_tick_interval() # Revert to initial rate if nothing critical and not too large

# 2. Adjust Focus (Attention)
if critical_manifestations:
# Focus on the sector of the most critical manifestation
sector_counts = {}
for m in critical_manifestations:
sector_counts[m.sector] = sector_counts.get(m.sector, 0) + 1
most_critical_sector = max(sector_counts, key=sector_counts.get) # Get sector with most critical items
self.scheduler.set_focus_sector(most_critical_sector)
elif new_emergencies:
# Focus on the sector where a new emergency emerged
self.scheduler.set_focus_sector(new_emergencies[0].sector)
elif high_priority_blueprints_pending:
# Focus on the sector of the highest priority pending blueprint
highest_prio_bp = sorted(high_priority_blueprints_pending, key=lambda b: b.priority, reverse=True)[0]
if highest_prio_bp.sector:
self.scheduler.set_focus_sector(highest_prio_bp.sector)
else:
self.scheduler.set_focus_sector(None) # Global focus if no specific sector
else:
self.scheduler.set_focus_sector(None) # No specific critical focus, revert to global

print("--- Starting Linguistic Simulation Theory Engine ---")
self.scheduler.start(self.process_tick)

self.scheduler.stop()

self.history = []        # For memory - stores changes over time

self.properties[key] = value
self._update_current_state_snapshot()

self.relationships[rel_type] = []
self.relationships[rel_type].append(target_id)
self._update_current_state_snapshot()

"""Internal helper to keep snapshot up-to-date. Subclasses will extend this."""
self.current_state_snapshot = self.properties.copy()

"timestamp": time.time(),
"tick": RealitySimulationEngine.current_tick, # Global tick from scheduler

"state": current_state_snapshot if current_state_snapshot is not None else self.current_state_snapshot.copy()
})

Conceptual embedding model. Replaces keyword matching with vector embeddings.
In a real system, this would be a large NLP model (e.g., Sentence Transformers, OpenAI embeddings).
For pseudocode, it's a simple hashing/keyword presence for demonstration purposes.

def generate_embedding(self, text: str) -> list[float]:
"""
Generates a conceptual vector embedding for a given text.
In a real system, this would use a robust NLP model like:
from sentence_transformers import SentenceTransformer
model = SentenceTransformer('all-MiniLM-L6-v2')
return model.encode(text).tolist()

Here, a simple "pseudo-embedding" based on character counts, length,
and presence of common "conceptual" words to simulate distinct vectors.
"""
if not text: return [0.0] * 9 # Return a zero vector for empty text

# A very simplistic placeholder for vector generation
embedding_vec = [
sum(ord(char) for char in text) / 10000.0, # Magnitude based on chars
len(text) / 100.0,                         # Length
1.0 if "concept" in text.lower() else 0.0,
1.0 if "system" in text.lower() else 0.0,
1.0 if "rule" in text.lower() else 0.0,
1.0 if "design" in text.lower() else 0.0,
1.0 if "physical" in text.lower() else 0.0,
1.0 if "applied" in text.lower() else 0.0,
random.random() * 0.01 # Small random component for variation, to prevent perfect matches often
]
return embedding_vec

def cosine_similarity(self, vec1: list[float], vec2: list[float]) -> float:
"""
Calculates cosine similarity between two conceptual vectors.
(dot product / (norm1 * norm2)).
"""
if not vec1 or not vec2 or len(vec1) != len(vec2):
return 0.0

dot_product = sum(a * b for a, b in zip(vec1, vec2))
norm_vec1 = sum(a * a for a in vec1) ** 0.5
norm_vec2 = sum(b * b for b in vec2) ** 0.5

if norm_vec1 == 0 or norm_vec2 == 0:
return 0.0 # Handle zero vectors

return dot_product / (norm_vec1 * norm_vec2)


class SemanticMatcher:
"""
Uses vector embeddings for robust semantic similarity calculations.
Replaces keyword-based logic for a more nuanced "conceptual understanding."
"""
def __init__(self):
self.embedding_model = EmbeddingModel()
self._cache = {} # Cache embeddings to avoid re-computation for efficiency

def _get_embedding(self, definition: BaseDefinition) -> list[float]:
"""Get or generate embedding for a definition."""
if definition.id not in self._cache:
text_content = definition.get_semantic_content()
self._cache[definition.id] = self.embedding_model.generate_embedding(text_content)
return self._cache[definition.id]

def calculate_similarity(self, def1: BaseDefinition, def2: BaseDefinition) -> float:
"""
Calculates semantic similarity between two definitions using their embeddings.
"""
embedding1 = self._get_embedding(def1)
embedding2 = self._get_embedding(def2)
return self.embedding_model.cosine_similarity(embedding1, embedding2)

def find_best_match(self, source_def: BaseDefinition, target_definitions: list[BaseDefinition], min_similarity: float = 0.6) -> BaseDefinition | None:
"""Finds the best matching definition from a list based on semantic similarity."""
best_match = None
highest_similarity = min_similarity
for target_def in target_definitions:
similarity = self.calculate_similarity(source_def, target_def)
if similarity > highest_similarity:
highest_similarity = similarity
best_match = target_def

# --- Priority / Energy Model ---
def _prioritize_processing(self): # MODIFIED for Goal Alignment & Conflict Resolution
"""
Adjusts processing order or intensity based on blueprint priority, GOALS, and CONFLICT_RESOLUTION needs.
"""
# Calculate a 'goal_alignment_score' for each blueprint
for bp in self.blueprints.values():
bp.add_property("_goal_alignment_score", 0.0) # Reset each tick
if bp.category == BlueprintCategory.GOAL:
bp.add_property("_goal_alignment_score", 1.0) # Goals themselves are highly aligned
bp.priority = max(bp.priority, 0.95) # Goals should generally have high effective priority

# Check if this blueprint is targeted by a GOAL (e.g., "Minimize Failures of X")
for goal_bp in [g for g in self.blueprints.values() if g.category == BlueprintCategory.GOAL]:
if goal_bp.properties.get("target_blueprint_id") == bp.id:
# If this blueprint is *the subject* of a goal, its priority might be boosted to give it attention
bp.add_property("_goal_alignment_score", bp.properties.get("_goal_alignment_score", 0.0) + 0.5)

# If it's a CONFLICT_RESOLUTION blueprint, boost its priority significantly
if bp.category == BlueprintCategory.CONFLICT_RESOLUTION:
bp.priority = max(bp.priority, 0.9) # High priority to resolve conflicts

# Sort by (original_priority + goal_alignment_score)
self.blueprints_by_priority = sorted(self.blueprints.values(),
key=lambda b: b.priority + b.properties.get("_goal_alignment_score", 0.0),
reverse=True)

# --- Constraint Evaluation ---
def _evaluate_constraints(self, manifestations: list[Manifestation]):
"""Checks MP and HA constraints to maintain system integrity (the 'grammar')."""
print("  Evaluating system constraints (MP, HA)...")
for manifestation in manifestations:
for constraint in self.constraints.values():

if constraint.type == ConstraintType.DIRECTIVE_LOGIC:
if constraint.name == "Melting Point" and "temperature" in manifestation.physical_attributes:
current_temp = manifestation.physical_attributes["temperature"]
if current_temp >= constraint.value and manifestation.physical_attributes.get("material_state") != "liquid":
manifestation.update_physical_state("material_state", "liquid")
print(f"    MP '{constraint.name}' applied: '{manifestation.name}' melts at {current_temp}°F.")

# HA Constraint (Integrity/Condition) - Triggers anomalies/crashes
if constraint.type == ConstraintType.INTEGRITY_CONDITION:
if constraint.name == "Heart Attack Threshold" and "cardiac_stress" in manifestation.physical_attributes:
if manifestation.physical_attributes["cardiac_stress"] >= constraint.value:
error_msg = f"HA '{constraint.name}' violated! '{manifestation.name}' experiences critical biological crash!"
print(f"    ERROR: {error_msg}")
manifestation.update_physical_state("system_status", "CRITICAL_ERROR: BIOLOGICAL_CRASH")
self._record_anomaly(manifestation, constraint, error_msg)
self._trigger_anomaly_response(manifestation, constraint)

if constraint.name == "High Altitude Threshold" and "altitude" in manifestation.physical_attributes:
current_altitude = manifestation.physical_attributes["altitude"]
if current_altitude > constraint.value and manifestation.physical_attributes.get("alert_status") != "HIGH_ALTITUDE_CRISIS":
error_msg = f"HA '{constraint.name}' violated! '{manifestation.name}' exceeds critical altitude ({current_altitude}ft)!"
print(f"    WARNING: {error_msg}")
manifestation.update_physical_state("alert_status", "HIGH_ALTITUDE_CRISIS")
self._record_anomaly(manifestation, constraint, error_msg)
self._trigger_anomaly_response(manifestation, constraint)

# --- NEW FEATURE: Multi-Manifestation Emergence & Identity Formation ---
def _detect_multi_manifestation_emergence(self): # MODIFIED for Identity Formation

current_emergence = "Economic Boom Phase"
elif avg_growth < -0.05 and avg_stability < 0.3:
current_emergence = "Economic Recession Phase"

if current_emergence:
if current_emergence not in self._persistent_emergences:
# Create or find EMERGENCE_PATTERN Blueprint

f"A pattern of {current_emergence.lower()}.", priority=0.9, sector=SimulationSector.ECONOMICS)
self.register_blueprint(emergent_bp)
existing_emergent_bp = emergent_bp

self._persistent_emergences[current_emergence] = {"last_tick": self.current_tick, "count": 1, "blueprint_id": existing_emergent_bp.id}

# Create Manifestation for this immediate emergence
if not any(m.name == current_emergence for m in self.manifestations.values() if self.current_tick - m.history[-1]['tick'] < 2 if m.history):
emergent_m = Manifestation(current_emergence, ManifestationCategory.OBSERVED_EMERGENCE,
existing_emergent_bp.id, SimulationSector.ECONOMICS, f"The current observed {current_emergence.lower()}.")

emergent_m.update_physical_state("status", "active")

[m.id for m in economic_manifestations], f"Observed avg growth: {avg_growth:.2f}, stability: {avg_stability:.2f}")

else: # Emergence is already being tracked
self._persistent_emergences[current_emergence]["count"] += 1
self._persistent_emergences[current_emergence]["last_tick"] = self.current_tick

# Identity Formation logic: If an emergence persists for enough ticks
persistence_threshold = 5 # How many consecutive ticks to be considered persistent
if self._persistent_emergences[current_emergence]["count"] >= persistence_threshold:
systemic_identity_name = f"{current_emergence.replace(' Phase', '')} Regime"
if not any(bp.name == systemic_identity_name for bp in self.blueprints.values() if bp.category == BlueprintCategory.SYSTEMIC_IDENTITY):

# --- NEW FEATURE: Blueprint Tension Detection ---
def _detect_blueprint_tension(self): # NEW
"""
Detects conflicts or contradictions between blueprints.
"""
print("  Detecting blueprint tension...")
current_tensions = []
for i, bp1 in enumerate(list(self.blueprints.values())):
for j, bp2 in enumerate(list(self.blueprints.values())):
if i >= j: continue # Avoid self-comparison and duplicate pairs
if bp1.id == bp2.id: continue # Should not happen with i>=j but good check

# Example 1: Contradictory Goals (simple property check)
if bp1.category == BlueprintCategory.GOAL and bp2.category == BlueprintCategory.GOAL:
if bp1.sector == bp2.sector and bp1.sector is not None: # Only if in the same domain
bp1_target = bp1.properties.get("target_metric")
bp2_target = bp2.properties.get("target_metric")
bp1_trend = bp1.properties.get("desired_trend")
bp2_trend = bp2.properties.get("desired_trend")

if bp1_target == bp2_target and bp1_trend == "increasing" and bp2_trend == "decreasing":
# Check if this tension was already recorded recently
if not any(t for t in self.tensions if t['blueprint1_id'] == bp1.id and t['blueprint2_id'] == bp2.id and self.current_tick - t['tick'] < 10):
self._record_tension(bp1, bp2, "Contradictory Goals", 0.9,
f"Both aim for {bp1_target} but with opposite trends ({bp1_trend} vs {bp2_trend}).")
current_tensions.append((bp1.id, bp2.id))


# Example 2: Semantic Conflict on related policies (Profit vs Environment)
if bp1.sector == SimulationSector.ECONOMICS and bp2.sector == SimulationSector.ECONOMICS:
if "profit" in bp1.name.lower() and "environmental" in bp2.name.lower():
# Using semantic similarity to detect "relatedness"
semantic_sim = self.semantic_matcher.calculate_similarity(bp1, bp2)
if semantic_sim > 0.3: # They are somewhat related conceptually
if not any(t for t in self.tensions if t['blueprint1_id'] == bp1.id and t['blueprint2_id'] == bp2.id and self.current_tick - t['tick'] < 10):
self._record_tension(bp1, bp2, "Policy Trade-off", 0.7,
f"Potential conflict between '{bp1.name}' (profit) and '{bp2.name}' (environment).")
current_tensions.append((bp1.id, bp2.id))


# Based on detected tensions, potentially create a CONFLICT_RESOLUTION blueprint
for tension_record in self.tensions:
# Only act on recent, high-severity tensions
if tension_record["severity"] > 0.8 and self.current_tick - tension_record["tick"] < 5:
bp1 = self.blueprints.get(tension_record['blueprint1_id'])
bp2 = self.blueprints.get(tension_record['blueprint2_id'])
if bp1 and bp2:
resolution_bp_name = f"Resolve_{bp1.name.replace(' ', '')}_vs_{bp2.name.replace(' ', '')}"
if not any(b.name == resolution_bp_name for b in self.blueprints.values() if b.category == BlueprintCategory.CONFLICT_RESOLUTION):
resolution_bp = Blueprint(
resolution_bp_name,
BlueprintCategory.CONFLICT_RESOLUTION,
f"Blueprint to mediate tension between '{bp1.name}' and '{bp2.name}' (Type: {tension_record['tension_type']}).",
priority=0.9,
sector=bp1.sector if bp1.sector == bp2.sector else None,
time_horizon=TimeHorizon.MEDIUM_TERM
)
resolution_bp.add_property("source_tension_ids", [bp1.id, bp2.id])
self.register_blueprint(resolution_bp)
print(f"      Conflict Resolution Blueprint created: '{resolution_bp.name}'")

# --- NEW FEATURE: Intent Drift & Goal Formation ---
def _apply_intent_drift(self): # MODIFIED for Goal Formation & Time Horizon
"""
Blueprints slowly mutate based on outcomes (success/failure rates).
Also, if persistent failures occur, new GOAL blueprints might be formed.
"""
print("  Applying intent drift to blueprints...")
for bp_id, blueprint in list(self.blueprints.items()):
# Ignore emergent patterns, recovery protocols, goals, conflict resolution, systemic identities for direct drift
if blueprint.category in [BlueprintCategory.EMERGENT_PATTERN, BlueprintCategory.CONCEPTUAL_FRAMEWORK,
BlueprintCategory.GOAL, BlueprintCategory.CONFLICT_RESOLUTION,
BlueprintCategory.SYSTEMIC_IDENTITY] or \
"RecoveryProtocolFor_" in blueprint.name:
continue

total_outcomes = blueprint.failure_count + blueprint.success_count

# Adjust interaction threshold based on time horizon
interaction_threshold = 5 # Default for SHORT_TERM
if blueprint.time_horizon == TimeHorizon.MEDIUM_TERM:
interaction_threshold = 10
elif blueprint.time_horizon == TimeHorizon.LONG_TERM:
interaction_threshold = 20 # Long-term blueprints require more data before drifting

if total_outcomes > interaction_threshold and self.current_tick % 5 == 0:
failure_rate = blueprint.failure_count / total_outcomes
old_priority = blueprint.priority

# Goal Formation: If blueprint is failing badly and consistently, create a goal to address it
if failure_rate > 0.8 and blueprint.failure_count > 10 and \
not any(g.category == BlueprintCategory.GOAL and f"Minimize {blueprint.name} Failures" in g.name for g in self.blueprints.values()):
goal_name = f"Minimize {blueprint.name} Failures"
new_goal_bp = Blueprint(
goal_name, BlueprintCategory.GOAL,
f"Strategic goal to reduce failures caused by '{blueprint.name}'.",
priority=1.0, sector=blueprint.sector, time_horizon=TimeHorizon.MEDIUM_TERM
)
new_goal_bp.add_property("target_blueprint_id", blueprint.id)
new_goal_bp.add_property("desired_trend", "decreasing_failure_rate")
self.register_blueprint(new_goal_bp)
self._record_intent_drift(bp_id, "goal_creation", "None", new_goal_bp.id,
f"Persistent high failure rate ({failure_rate:.2f}) led to new goal.")
print(f"    DRIFT (GOAL): New Goal created: '{new_goal_bp.name}'.")

# Priority Decrease (after potential goal creation)
if failure_rate > 0.6:
new_priority = max(0.1, round(blueprint.priority - 0.1, 2))
if new_priority != old_priority:
blueprint.priority = new_priority
self._record_intent_drift(bp_id, "priority_decrease", old_priority, new_priority,
f"High failure rate ({failure_rate:.2f})")
print(f"    DRIFT: Blueprint '{blueprint.name}' priority decreased to {new_priority:.2f} due to high failure rate.")
if not blueprint.properties.get("needs_re_evaluation"):
blueprint.update_blueprint_property("needs_re_evaluation", True)
self._record_intent_drift(bp_id, "add_property", "None", "needs_re_evaluation:True",
f"High failure rate ({failure_rate:.2f})")
elif failure_rate < 0.2 and blueprint.success_count > interaction_threshold: # If mostly succeeding
new_priority = min(1.0, round(blueprint.priority + 0.05, 2))
if new_priority != old_priority:
blueprint.priority = new_priority
self._record_intent_drift(bp_id, "priority_increase", old_priority, new_priority,
f"High success rate ({(1-failure_rate):.2f})")
print(f"    DRIFT: Blueprint '{blueprint.name}' priority increased to {new_priority:.2f} due to high success rate.")
if blueprint.properties.get("needs_re_evaluation"):
old_value = blueprint.properties["needs_re_evaluation"]
del blueprint.properties["needs_re_evaluation"]
blueprint._update_current_state_snapshot()
self._record_intent_drift(bp_id, "remove_property", f"needs_re_evaluation:{old_value}", "None",
f"Consistently successful ({(1-failure_rate):.2f})")


# --- NEW FEATURE: Narrative Memory Layer ---
def _synthesize_narratives(self): # NEW
"""
Periodically converts raw logs into abstracted narratives, simulating 'reflection'.
"""
if self.current_tick % 10 != 0: # Synthesize narratives every 10 ticks
return

print("  Synthesizing narratives...")
recent_anomalies = [a for a in self.historical_records["anomaly_logs"] if a["tick"] > self.current_tick - 10]
recent_emergences = [e for e in self.historical_records["emergence_logs"] if e["tick"] > self.current_tick - 10]
recent_intent_drift = [d for d in self.historical_records["intent_drift_logs"] if d["tick"] > self.current_tick - 10]
recent_tensions = [t for t in self.historical_records["tension_logs"] if t["tick"] > self.current_tick - 10]

# Narrative 1: Systemic Instability
if len(recent_anomalies) > 3: # If many anomalies recently
involved_bps = set(a["manifestation_state"].get("originating_blueprint_id") for a in recent_anomalies if a["manifestation_state"])
involved_entities = [m_id for a in recent_anomalies for m_id in [a["manifestation_id"]]]
involved_entities.extend([bp_id for bp_id in involved_bps if bp_id])

narrative = NarrativeState(
summary="Systemic instability observed with frequent constraint violations across sectors.",
involved_entities=list(set(involved_entities)),
confidence=0.8,
sentiment="alarming",
category="Crisis"
)
self._record_narrative(narrative)

# Narrative 2: Emergent Pattern of Growth/Decline
for emergence in recent_emergences:
if emergence["category"] == ManifestationCategory.OBSERVED_EMERGENCE.name and \
("Boom" in emergence["name"] or "Recession" in emergence["name"] or "Distress" in emergence["name"] or "Regime" in emergence["name"]):
narrative = NarrativeState(
summary=f"Persistent {emergence['name']} detected. Details: {emergence['details']}",
involved_entities=emergence["involved_manifestations"],
confidence=0.9,
sentiment="positive" if "Boom" in emergence["name"] else "negative",
category="PatternShift"
)
self._record_narrative(narrative)

# Narrative 3: Blueprint Adaptation
for drift in recent_intent_drift:
bp_name = self.blueprints.get(drift['blueprint_id'], "Unknown Blueprint").name
if "priority_decrease" in drift['change_type']:
narrative = NarrativeState(
summary=f"Blueprint '{bp_name}' is adapting, exhibiting reduced priority due to past failures.",
involved_entities=[drift['blueprint_id']],
confidence=0.7,
sentiment="neutral",
category="Adaptation"
)
self._record_narrative(narrative)
elif "goal_creation" in drift['change_type']:
goal_bp = self.blueprints.get(drift['new_value'])
if goal_bp:
narrative = NarrativeState(
summary=f"System initiated new strategic goal: '{goal_bp.name}' to address persistent issues.",
involved_entities=[drift['blueprint_id'], goal_bp.id],
confidence=0.9,
sentiment="improving",
category="StrategyChange"
)
self._record_narrative(narrative)

# Narrative 4: Contradiction/Tension
if len(recent_tensions) > 0:
for tension in recent_tensions:
bp1_name = self.blueprints.get(tension['blueprint1_id'], "Unknown BP").name
bp2_name = self.blueprints.get(tension['blueprint2_id'], "Unknown BP").name
narrative = NarrativeState(
summary=f"Internal tension detected between '{bp1_name}' and '{bp2_name}'. Type: {tension['tension_type']}.",
involved_entities=[tension['blueprint1_id'], tension['blueprint2_id']],
confidence=tension['severity'],
sentiment="alarming",
category="Conflict"
)
self._record_narrative(narrative)


# --- NEW FEATURE: Feedback into the Scheduler (Time Awareness / Attention) ---
def _assess_system_load_and_focus(self): # MODIFIED for narratives, goals, tension, time_horizon
"""
Adjusts scheduler parameters (tick rate, focus) based on system state,
including narratives, goals, and tensions.
"""
print("  Assessing system load and focus...")
total_anomalies = len([a for a in self.historical_records["anomaly_logs"] if a["tick"] == self.current_tick])
critical_manifestations = [m for m in self.manifestations.values() if "CRITICAL_ERROR" in str(m.physical_attributes.get("system_status")) or "HIGH_ALTITUDE_CRISIS" == str(m.physical_attributes.get("alert_status"))]
high_priority_blueprints_pending = [b for b in self.blueprints.values() if b.priority > 0.8 and b.properties.get("manifested_status") != "complete"]
new_emergencies = [m for m in self.manifestations.values() if m.category == ManifestationCategory.OBSERVED_EMERGENCE and m.physical_attributes.get("status") == "critical" and m.history and m.history[-1]['tick'] == self.current_tick]
active_goals = [bp for bp in self.blueprints.values() if bp.category == BlueprintCategory.GOAL]
high_tension_situations = [t for t in self.tensions if t["severity"] > 0.7 and self.current_tick - t["tick"] < 10] # Recent, high-severity tensions
recent_alarming_narratives = [n for n in self.narratives if n.sentiment == "alarming" and self.current_tick - n.tick < 10]
recent_strategy_narratives = [n for n in self.narratives if n.category == "StrategyChange" and self.current_tick - n.tick < 10]


# 1. Adjust Tick Rate (Urgency / Time Awareness)
# Speed up if there are critical issues, new emergencies, high tension, or alarming narratives
if total_anomalies > 0 or len(critical_manifestations) > 0 or len(new_emergencies) > 0 or \
len(high_tension_situations) > 0 or len(recent_alarming_narratives) > 0:
if self.scheduler.tick_interval > 0.01:
self.scheduler.adjust_tick_interval(max(0.01, self.scheduler.tick_interval * 0.7)) # Accelerate by 30% for high urgency
# Slow down if the system is very large and calm, or when focusing on long-term goals/planning
elif len(self.manifestations) > 50 and len(self.blueprints) > 50 and self.current_tick % 10 == 0:
# Check for predominantly LONG_TERM goals or recent strategic shifts (implying planning)
long_term_goal_count = sum(1 for g in active_goals if g.time_horizon == TimeHorizon.LONG_TERM)
if long_term_goal_count > len(active_goals) / 2 or len(recent_strategy_narratives) > 0:
if self.scheduler.tick_interval < 0.5:
self.scheduler.adjust_tick_interval(min(0.5, self.scheduler.tick_interval * 1.5)) # Decelerate by 50% for deep thought/planning
elif self.scheduler.tick_interval < 0.5:
self.scheduler.adjust_tick_interval(min(0.5, self.scheduler.tick_interval * 1.2)) # Decelerate by 20%
else:
self.scheduler.reset_tick_interval() # Revert to initial rate if nothing critical and no specific planning focus

# 2. Adjust Focus (Attention)
if critical_manifestations:
sector_counts = {}
for m in critical_manifestations:
sector_counts[m.sector] = sector_counts.get(m.sector, 0) + 1
most_critical_sector = max(sector_counts, key=sector_counts.get)
self.scheduler.set_focus_sector(most_critical_sector)
elif new_emergencies:
self.scheduler.set_focus_sector(new_emergencies[0].sector)
elif high_tension_situations:
# Focus on a sector involved in a high-tension situation or the conflict resolution blueprint
bp1 = self.blueprints.get(high_tension_situations[0]['blueprint1_id'])
if bp1 and bp1.sector:
self.scheduler.set_focus_sector(bp1.sector)
elif active_goals:
# Prioritize focus on the sector of the highest priority active goal
highest_prio_goal = sorted(active_goals, key=lambda g: g.priority, reverse=True)[0]
if highest_prio_goal.sector:
self.scheduler.set_focus_sector(highest_prio_goal.sector)
elif high_priority_blueprints_pending:
highest_prio_bp = sorted(high_priority_blueprints_pending, key=lambda b: b.priority, reverse=True)[0]
if highest_prio_bp.sector:
self.scheduler.set_focus_sector(highest_prio_bp.sector)
else:
self.scheduler.set_focus_sector(None)
else:
self.scheduler.set_focus_sector(None)

Key Changes and Explanations:
New Enums:
BlueprintCategory: Added GOAL, CONFLICT_RESOLUTION, SYSTEMIC_IDENTITY.
TimeHorizon: SHORT_TERM, MEDIUM_TERM, LONG_TERM for directional time awareness.
Blueprint Class Modifications:
Added time_horizon attribute, defaulting to SHORT_TERM. This informs intent drift and scheduler assessment.
NarrativeState Class (NEW):
A new data structure to hold "compressed meaning" from historical events. Includes summary, involved_entities, confidence, sentiment, category, and cause_bp_id.
RealitySimulationEngine __init__ Additions:
self.narratives: A list to store generated NarrativeState objects.
self.tensions: A list to store records of detected blueprint conflicts.
self._persistent_emergences: A dictionary to track the duration and frequency of emergent patterns, crucial for forming SYSTEMIC_IDENTITY.
historical_records: Extended to include "narrative_logs" and "tension_logs".
New Logging Methods (_record_narrative, _record_tension):
These methods formally log the creation of narratives and the detection of tensions within the system's history.
process_tick() Order Adjustment:
The new cognitive steps (_detect_multi_manifestation_emergence, _detect_blueprint_tension, _apply_intent_drift, _synthesize_narratives, _assess_system_load_and_focus) are called in a specific order to ensure information flows logically (e.g., emergence leads to identity, drift leads to goals, logs are processed into narratives, and then all these inform attention).
_synthesize_narratives() (NEW):
Runs periodically (every 10 ticks in this example) to simulate "reflection."
Analyzes recent anomaly_logs, emergence_logs, intent_drift_logs, and tension_logs.
Generates NarrativeState objects for significant events like systemic instability, persistent patterns, blueprint adaptation, new goal formation, and detected conflicts.
_detect_blueprint_tension() (NEW):
Iterates through blueprints to find conflicts.
Contradictory Goals: Detects when two GOAL blueprints in the same sector aim for opposite trends on the same metric.
Policy Trade-off: Uses SemanticMatcher to find conceptually related policies (e.g., maximize profit vs. minimize environmental impact) within the same sector.
Conflict Resolution Blueprint: If a high-severity tension is detected, it registers a new CONFLICT_RESOLUTION blueprint, which gains high priority to address the conflict.
_detect_multi_manifestation_emergence() (MODIFIED for Identity Formation):
Persistence Tracking: Now uses self._persistent_emergences to count how many consecutive ticks an emergent pattern (like "Economic Boom Phase" or "Widespread Ecosystem Distress") is observed.
Systemic Identity Formation: If an emergent pattern persists beyond a persistence_threshold (e.g., 5 ticks), a new SYSTEMIC_IDENTITY blueprint is registered. This represents the system's "belief" that this pattern is a fundamental, persistent state (a "regime").
_apply_intent_drift() (MODIFIED for Goal Formation & Time Horizon):
Time Horizon Influence: The interaction_threshold (how many successes/failures are needed before drift occurs) is now adjusted based on the Blueprint's time_horizon (e.g., long-term blueprints drift slower).
Goal Formation: If a blueprint consistently experiences a very high failure rate over a significant number of interactions, the system generates a new GOAL blueprint (e.g., "Minimize X Failures") to strategically address that issue.
_prioritize_processing() (MODIFIED for Goal Alignment & Conflict Resolution):
Goal Alignment Score: Blueprints now get a _goal_alignment_score. GOAL blueprints themselves have a high score, and blueprints whose IDs are targeted by an active GOAL (e.g., "Minimize X Failures" targeting blueprint X) also get a boost.
Conflict Resolution Priority: CONFLICT_RESOLUTION blueprints are given high priority to ensure internal conflicts are addressed.
The sorting now considers this combined priority.
_assess_system_load_and_focus() (MODIFIED for Cognitive Input):
Tick Rate Adjustment (Urgency & Strategic Foresight):
Still speeds up for critical anomalies, emergencies, high tension, or alarming narratives.
Now slows down not just for large, calm systems, but also if there are many active LONG_TERM goals or recent StrategyChange narratives, simulating "deep thought" or long-term planning.
Focus Adjustment (Attention):
Focus priorities: Critical manifestations > New emergencies > High-tension situations > Active GOAL sectors > High-priority blueprints. This simulates the system directing its "attention" to what is most urgent or strategically important based on its internal state and goals.
Example Usage (if __name__ == "__main__":):

Tension Demo: Added bp_maximize_profit and bp_minimize_env_impact in the MANUFACTURING sector, designed to trigger a "Policy Trade-off" tension.

Summary: New sections added to print out the Narrative Log, Tension Log, Systemic Identities Formed, Active Goals, and Conflict Resolution Blueprints at the end of the simulation.
